# Step 0: Get user input for documents
documents = {}
print("Enter your documents (type 'done' when finished):")

doc_id = 1
while True:
    text = input(f"Document {doc_id}: ")
    if text.lower() == 'done':
        break
    documents[doc_id] = text
    doc_id += 1

# Step 1: Identify the terms in each document and its frequency
print("\nStep 1: Term, DocID")
term_doc_list = []

for doc_id, text in documents.items():
    terms = text.split()
    for term in terms:
        term_doc_list.append((term, doc_id))
        print(f"{term:<10} {doc_id}")

# Step 2: Arrange the list in ascending order (alphabetical order of terms)
print("\nStep 2: Sorted Term, DocID")
sorted_terms = sorted(term_doc_list)
for term, doc_id in sorted_terms:
    print(f"{term:<10} {doc_id}")

# Step 3: Identify unique terms across documents and list their DocIDs
print("\nStep 3: Unique Terms with DocIDs")
unique_terms = {}

for term, doc_id in sorted_terms:
    if term not in unique_terms:
        unique_terms[term] = []
    if doc_id not in unique_terms[term]:
        unique_terms[term].append(doc_id)

for term in unique_terms:
    print(f"{term:<10} {unique_terms[term]}")

# Step 4: Create Posting Lists
print("\nStep 4: Posting List")
for term in sorted(unique_terms):
    doc_ids = " -> ".join(str(id) for id in sorted(unique_terms[term]))
    print(f"{term:<10} -> {doc_ids}")
