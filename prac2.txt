Q.1 Using properties in class component 
CODE:  
 
P2aClassprop.js  
import React from "react";  
import { Component } from "react";  
class Parent extends Component{  
 render()  
 {  
 return(  
 <div>  
 <h1>This is Parent Class {this.props.address} 
</h1>  <Child p_name="Parent Name" p_age = {50}/>  
</div>  
 );  
 }  
}  
class Child extends Component{  
 render()  
 {  
 return(  
 <div>  
 <h2>  
 {this.p_name} {this.props.p_age}  </h2>  
 </div>  
 );  
 }  
}  
export {Parent, Child} 
 
App.js  
import {Parent, Child} from "./P2aClassprop.js";  
function App()  
{  
 return(  
 <div>  
 <Parent address="qwerty"/>  
 <Child/>  
 </div>  
 );  
}  
export default App;

Q.3 Updating message using state class component. 
CODE:  
 
Stateclass.js 
import React from "react";  
import { Component } from "react";  
class Message extends Component{  
 constructor()  
 {  
 super()  
 this.state={  
 message:"Welcome Visitors"  
 }  
 }  
render()  
{  
 return(  
 <div>  
 <h1>{this.state.message}</h1>  
 <button onClick={()=>this.changeMessage()}>Click here</button>  
</div>  
 )  
}  
changeMessage()  
{  
 this.setState(  
 {  
 message:"Thank You for Visiting!"  
 }  
 )  
}  
}  
export default Message  
 
App.js 
import Message from "./Stateclass.js";  
function App(){  
 return(  
 <div className="App">  
 <Message/>  
 </div>  
);  
}  
export default App;

Q.4 Create a counter using state class component 
CODE:  
 
P2Counter.js  
import React from "react";  
import { Component } from "react"; 
class Message extends Component{  
constructor()  
{  
super()  
this.state={  
Message:"Welcome"  
}  
}  
render(){  
return(  
<div>  
<h1>{this.state.Message}</h1>  
<button onClick={()=>this.changeMessage()}>Click Here</button>  </div>  
)  
}

changeMessage()  
{  
this.setState(  
{  
Message: "Thank you for visiting!!"  
}   
)  
}  
}  
export default Message  
 
App.js   
import './App.css';  
import Message from './P1Q3'; 
function App() {  
return (  
<div className="App">  
<Message/>  
</div>  
);  
}  
export default App;
