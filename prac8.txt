Aim: Fetch the API data and render the list on React App. 
UserPosts.js 
import React, { useState, useEffect } from 'react'; 
 
function P6iv() { 
    const [posts, setPosts] = useState([]); 
 
    useEffect(() => { 
        const fetchData = async () => { 
            const response = await fetch('http://jsonplaceholder.typicode.com/posts'); 
            const data = await response.json();  // Ensure you're using `const data` here 
            setPosts(data); 
        }; 
        fetchData(); 
    }, []);  // Correctly pass an empty dependency array 
 
    return ( 
        <div> 
            {posts.map((post) => ( 
                <div key={post.id}> 
                    <p>{post.id}</p> 
                    <br /> 
                    <h3>Title: {post.title}</h3> 
                    <p>Body: {post.body}</p> 
                </div> 
            ))} 
        </div> 
    );

} 
 
export default P6iv; 
 
App.js 
import React from 'react'; 
import './App.css'; 
import P6iv from './UserPosts'; 
function App() { 
  return ( 
    <div className="App"> 
      <header className="App-header"> 
        <P6iv/> 
      </header>  
    </div> 
  ); 
} 
export default App;
